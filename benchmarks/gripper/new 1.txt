(define (problem robot-object-move)
   (:domain own) 
   (:objects r1 r2 r3 r4 r5 r6 r7 r8 r9 ob1 ob2 ob3 left right up down) 
   (:init (room r1)
		  (room r2)
		  (room r3)
		  (room r4)
		  (room r5)
		  (room r6)
		  (room r7)
		  (room r8)
		  (room r9)
		  (object ob1)
		  (object ob2)
		  (object ob3)
		  (robot left)
		  (robot right)
		  (robot up)
		  (robot down)
		  
   (:goal (and (at ob1 r7)
               (at ob2 r2)
               (at ob3 r9)))))
